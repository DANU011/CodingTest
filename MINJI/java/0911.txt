- 나머지가 1이 되는 수 찾기
class Solution {
    public int solution(int n) {
        int answer = 0;
        for(int i = 2; i < n; i++) {
            if(n % i == 1) {
                answer = i;
                break;
            }
        }
        return answer;
    }
}

- 평균 구하기
class Solution {
    public double solution(int[] arr) {
        double answer = 0;
        double result = 0;
        for(int i = 0; i < arr.length; i++) {
            result += arr[i];
        }
        answer = result / arr.length;
        return answer;
    }
}
