- 자연수 뒤집어 배열로 만들기
class Solution {
    public int[] solution(long n) {
        String a = "" + n;
        int[] answer = new int[a.length()];
        int cnt=0;
        while(n > 0) {
            answer[cnt]=(int)(n%10);
            n/=10;
            System.out.println(n);
            cnt++;
        }
        return answer;
    }
}

- 문자열을 정수로 바꾸기
class Solution {
    public int solution(String s) {
        return Integer.parseInt(s);
    }
}
